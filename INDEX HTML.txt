<script>
    // MASUKKAN CONFIG FIREBASE ANDA DI SINI (Dapatkan dari Firebase Console)
    const firebaseConfig = {
        databaseURL: "URL_FIREBASE_ANDA",
    };
    
    firebase.initializeApp(firebaseConfig);
    const database = firebase.database();

    const statusRef = database.ref('LecturerStatus');
    statusRef.on('value', (snapshot) => {
        const val = snapshot.val();
        const display = document.getElementById('status-display');
        display.innerText = val ? val : "TIADA";
        
        // Tukar warna berdasarkan teks
        display.className = ""; // Reset
        if(val == "Hadir") display.classList.add("status-active");
        else if(val == "Busy") display.classList.add("status-busy");
        else if(val == "Meeting") display.classList.add("status-meeting");
        else display.classList.add("status-none");
    });
</script>
